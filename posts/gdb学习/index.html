// for_breakpoint.cpp
#include <iostream>
using namespace std;  

void my_func1(int i) {
  std::cout << "Now in func1, i = " << i << "\n";
}

void my_func2(int i) {
  std::cout << "Now in func2, i = " << i << "\n";
}

int main() {
  for (int i = 0; i < 3; ++i) {
    my_func1(i);
    my_func2(i);
  }

  return 0;
}
